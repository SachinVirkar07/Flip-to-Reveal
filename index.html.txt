<script>
const topics = [
  "🚀 SpaceX milestone",
  "📜 Historical event",
  "💡 Startup idea",
  "🧠 Psychology fact",
  "🌍 Geography trivia",
  "📱 Tech update",
  "🎬 Movie fact",
  "📚 Book recommendation",
  "⚽ Sports highlight",
  "🔬 Science discovery",
  "🎨 Art inspiration",
  "💬 Quote of the day"
];

// Shuffle topics daily
function shuffle(array) {
  return array.sort(() => Math.random() - 0.5);
}

const shuffled = shuffle([...topics]);

const grid = document.getElementById("grid");

shuffled.forEach((topic, index) => {
  const card = document.createElement("div");
  card.className = "card";
  card.innerText = "Click to Reveal";

  card.onclick = () => {
    card.innerText = topic;
    card.classList.add("revealed");
  };

  grid.appendChild(card);
});
</script>
